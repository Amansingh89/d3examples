Data Driven Documents

D3 allows document elements to be generated from data. Hence the name Data Driven Documents (D3).

The circles plotted above are generated using scalable vector graphics (svg) and based on a simple array, which is passed through D3 to generate the necessary SVG elements.

    var w = 1000;
    var h = 200;

    data = [50, 60, 70, 30, 20, 10, 20, 20, 30, 40];
    var colors = d3.scale.category10();

    //Create SVG element
    var svg = d3.select("#draw_here")
            .append("svg")
            .attr("width", w)
            .attr("height", h);

    svg.selectAll("circle")
            .data(data)
            .enter()
            .append("circle")
            .attr("cx", function (d, i) {
                return 75 + (i * 80);
            })
            .attr("cy", h / 2)
            .attr("r", function (d, i) {
                return d;
            })
            .attr("fill", function(d, i) {
                return colors(i);
            })
            .attr("opacity", "0.5");
